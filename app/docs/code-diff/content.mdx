CodeDiff renders syntax-highlighted diffs with additions and deletions marked in color. Pass `oldCode` / `newCode` for file comparisons or a `patch` string for unified diffs from version control.

**Role:** Information. For displaying data the user reads. See [Design Guidelines](/docs/design-guidelines) for how component roles work.

## Getting Started

<Steps>

<Step title="Install the component">

Run this once from your project root.

```bash
npx shadcn@latest add @tool-ui/code-diff
```

</Step>

<Step title="Install the peer dependency">

CodeDiff uses [`@pierre/diffs`](https://www.npmjs.com/package/@pierre/diffs) for diff computation and rendering.

```bash
pnpm add @pierre/diffs
```

</Step>

<Step title="Use it in your application">

Render `CodeDiff` in your UI with tool-compatible props.

<div className="[&_pre]:max-h-[30rem] [&_pre]:overflow-y-auto">

```tsx
import { CodeDiff } from "@/components/tool-ui/code-diff";

export function MyComponent() {
  return (
    <CodeDiff
      id="my-diff-example"
      language="typescript"
      filename="lib/auth.ts"
      oldCode={`function checkAuth(token: string) {
  if (!token) throw new Error("Missing token");
  return jwt.verify(token, SECRET);
}`}
      newCode={`function checkAuth(token: string): AuthResult {
  if (!token) return { ok: false, error: "missing_token" };
  const decoded = jwt.verify(token, SECRET);
  return { ok: true, user: decoded };
}`}
    />
  );
}
```

</div>

</Step>

<Step title="Register it in your runtime provider">

Register this renderer so tool results display as `CodeDiff`.

```tsx
import { type Toolkit } from "@assistant-ui/react";
import { CodeDiff } from "@/components/tool-ui/code-diff";
import { safeParseSerializableCodeDiff } from "@/components/tool-ui/code-diff/schema";
import { createResultToolRenderer } from "@/components/tool-ui/shared";

export const toolkit: Toolkit = {
  showCodeDiff: {
    type: "backend",
    render: createResultToolRenderer({
      safeParse: safeParseSerializableCodeDiff,
      render: (parsedResult) => (
        <CodeDiff {...parsedResult} />
      ),
    }),
  },
};
```

</Step>

</Steps>

## Key Features

<FeatureGrid>
  <Feature icon="Code" title="Two input modes">
    Pass `oldCode`/`newCode` for file diffs, or `patch` for unified diffs
  </Feature>
  <Feature icon="Palette" title="Syntax highlighting">
    Language-aware highlighting powered by @pierre/diffs
  </Feature>
  <Feature icon="Expand" title="Split view">
    Side-by-side comparison with `diffStyle="split"`
  </Feature>
  <Feature icon="FoldVertical" title="Collapsible diffs">
    Collapse long diffs with `maxCollapsedLines`
  </Feature>
</FeatureGrid>
