A bare audio URL gives your users nothing to see. Audio wraps the browser's native player with artwork, title, duration, and source attribution so the track looks intentional rather than dumped into the chat.

## Getting Started

<Steps>

<Step title="Install the component">

Run this once from your project root.

```bash
npx shadcn@latest add @tool-ui/audio
```

</Step>

<Step title="Use it in your application">

Render `Audio` in your UI with tool-compatible props.

<div className="[&_pre]:max-h-[30rem] [&_pre]:overflow-y-auto">

```tsx
import { Audio } from "@/components/tool-ui/audio";

export function Example() {
  return (
    <Audio
      id="audio-example"
      assetId="sample-audio"
      src="https://samplelib.com/lib/preview/mp3/sample-6s.mp3"
      title="Bell Labs hallway recording"
      description="Ambient sounds where UNIX, C, and more took shape."
      artwork="https://images.unsplash.com/photo-1454165205744-3b78555e5572"
      durationMs={30000}
    />
  );
}
```

</div>

</Step>

<Step title="Register it in your runtime provider">

Register this renderer so tool results display as `Audio`.

```tsx
import { type Toolkit } from "@assistant-ui/react";
import { Audio } from "@/components/tool-ui/audio";
import { createResultToolRenderer } from "@/components/tool-ui/shared";

export const toolkit: Toolkit = {
  showAudio: {
    type: "backend",
    render: createResultToolRenderer({
      render: (parsedResult) => <Audio {...parsedResult} />,
    }),
  },
};
```

</Step>

</Steps>

## Key Features

<FeatureGrid>
  <Feature icon="Headphones" title="Native audio controls">
    Browser-native play, pause, seek, and volume
  </Feature>
  <Feature icon="Image" title="Artwork display">
    Album art or thumbnail displayed beside the player
  </Feature>
  <Feature icon="FileText" title="Metadata display">
    Title, description, duration, and file size at a glance
  </Feature>
  <Feature icon="Link" title="Source attribution">
    Artist, album, or source with an optional link
  </Feature>
</FeatureGrid>
