Add a tool-ui component like option-list to your application:

<QuickStartInstallTabs componentId="option-list" />

Your frontend code should look something like this:

```tsx
"use client";

import { lastAssistantMessageIsCompleteWithToolCalls } from "ai";
import { AssistantRuntimeProvider, Tools, useAui } from "@assistant-ui/react";
import { AssistantChatTransport, useChatRuntime } from "@assistant-ui/react-ai-sdk";
import { Thread } from "@/components/assistant-ui/thread";
import { type Toolkit } from "@assistant-ui/react";
import {
  OptionList,
  OptionListErrorBoundary,
  SerializableOptionListSchema,
  type OptionListSelection,
} from "@/components/tool-ui/option-list";

export const toolkit: Toolkit = {
  selectOptionFromListTool: {
    description: "Let the user choose options from a list.",
    parameters: SerializableOptionListSchema,
    render: ({ args, result, status, addResult, toolCallId }) => {
      if (status.type === "complete" && result) return `Selected: ${JSON.stringify(result)}`;
      if (status.type === "running") return <div>Loading...</div>;
      return (
        <OptionListErrorBoundary>
          <OptionList
            {...args}
            value={undefined}
            choice={result as OptionListSelection}
            onConfirm={(selection) => addResult(selection)}
          />
        </OptionListErrorBoundary>
      );
    },
  },
};

export const Assistant = () => {
  const runtime = useChatRuntime({
    transport: new AssistantChatTransport({ api: "/api/chat" }),
    sendAutomaticallyWhen: lastAssistantMessageIsCompleteWithToolCalls,
  });
  const aui = useAui({ tools: Tools({ toolkit }) });

  return (
    <AssistantRuntimeProvider runtime={runtime} aui={aui}>
      <div className="h-dvh">
        <Thread />
      </div>
    </AssistantRuntimeProvider>
  );
};
```

Your API chat route should look something like this:

```tsx
import { openai } from "@ai-sdk/openai";
import { streamText, convertToModelMessages, type UIMessage } from "ai";
import { frontendTools } from "@assistant-ui/react-ai-sdk";

export async function POST(req: Request) {
  const { messages, tools }: { messages: UIMessage[]; tools: Parameters<typeof frontendTools>[0] } = await req.json();

  const result = streamText({
    model: openai("gpt-4o"),
    messages: await convertToModelMessages(messages),
    tools: { ...frontendTools(tools) },
  });

  return result.toUIMessageStreamResponse();
}
```
