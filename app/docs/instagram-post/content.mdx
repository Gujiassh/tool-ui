## Source and Install

Install this component directly from the Tool UI shadcn registry:

```bash
npx shadcn@latest add https://tool-ui.com/r/instagram-post.json
```

This command installs `components/tool-ui/instagram-post`, `components/tool-ui/shared`, and `lib/ui/cn.ts` into your project.

## Usage

```tsx
import { type Toolkit } from "@assistant-ui/react";
import {
  InstagramPost,
  safeParseSerializableInstagramPost,
} from "@/components/tool-ui/instagram-post";
import { createResultToolRenderer } from "@/components/tool-ui/shared";

export const toolkit: Toolkit = {
  showInstagramPost: {
    type: "backend",
    render: createResultToolRenderer({
      safeParse: safeParseSerializableInstagramPost,
      render: (parsedResult) => <InstagramPost post={parsedResult} />,
    }),
  },
};
```
